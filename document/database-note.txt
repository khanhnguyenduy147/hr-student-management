CREATE TABLE Staff (
    staffId          NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    fullName         VARCHAR2(100 CHAR),
    mobileNumber     VARCHAR2(15 CHAR),
    referrerId       NUMBER,  -- Foreign key, tham chiếu đến một bảng khác, Id của người giới thiệu(đối tác)
    dateOfBirth      DATE,
    identifyNumber   VARCHAR2(20 CHAR),  -- Số CMND hoặc căn cước công dân
    initialCourse    VARCHAR2(50 CHAR),  -- Tên khóa học của ứng viên, mỗi ứng viên chỉ có một khóa khởi tạo nhưng có thể tham gia nhiều khóa đào tạo sau đó
    address          VARCHAR2(255 CHAR),
    hometown         VARCHAR2(100 CHAR),
    contact          VARCHAR2(15 CHAR),  -- Số liên hệ (nếu có khác với mobileNumber)
    email            VARCHAR2(100 CHAR) UNIQUE,
    studentStaffId   NUMBER,  -- Foreign key, tham chiếu đến một bảng khác, Tham chiếu đến bảng học viên nếu ứng viên bắt đầu từ học viên lên
    contractCode     VARCHAR2(50 CHAR),  -- Mã hợp đồng
    status           VARCHAR2(20 CHAR)  -- Tình trạng
);

-- Thêm các ràng buộc khóa ngoại cho referrerId và studentStaffId (nếu cần)
ALTER TABLE Staff ADD CONSTRAINT fk_referrer FOREIGN KEY (referrerId) REFERENCES ReferrerTable(id);  -- Thay 'ReferrerTable' với tên bảng thực tế
ALTER TABLE Staff ADD CONSTRAINT fk_studentStaff FOREIGN KEY (studentStaffId) REFERENCES StudentStaffTable(id);  -- Thay 'StudentStaffTable' với tên bảng thực tế
